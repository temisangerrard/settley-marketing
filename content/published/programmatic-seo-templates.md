## Technical Implementation

### For Next.js/React

**File: `/pages/invest/[city].tsx`**

```typescript
import { GetStaticPaths, GetStaticProps } from 'next'

interface CityData {
  name: string
  avgPrice: string
  yield: string
  appreciation: string
  vacancy: string
  reason1: string
  reason2: string
  reason3: string
}

export default function InvestInCity({ city }: { city: CityData }) {
  return (
    <div>
      <h1>Invest in {city.name} Real Estate Online</h1>
      {/* Use template above, replacing [VARIABLES] with city.* */}
    </div>
  )
}

export const getStaticPaths: GetStaticPaths = async () => {
  const cities = ['london', 'madrid', 'austin', /* ... */]
  return {
    paths: cities.map(city => ({ params: { city } })),
    fallback: false
  }
}

export const getStaticProps: GetStaticProps = async ({ params }) => {
  const cityData = getCityData(params?.city as string)
  return { props: { city: cityData } }
}
```
